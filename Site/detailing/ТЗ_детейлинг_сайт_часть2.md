```javascript
// Progressive loading для 3D
const loadCarModel = async () => {
	// 1. Загрузка low-poly версии (50KB)
	const lowPolyModel = await loadGLTF('/models/car-low.glb')
	scene.add(lowPolyModel)

	// 2. Фоновая загрузка high-poly (2MB)
	const highPolyModel = await loadGLTF('/models/car-high.glb')

	// 3. Плавная замена
	gsap.to(lowPolyModel.material, {
		opacity: 0,
		duration: 0.5,
		onComplete: () => {
			scene.remove(lowPolyModel)
			scene.add(highPolyModel)
			gsap.fromTo(
				highPolyModel.material,
				{ opacity: 0 },
				{ opacity: 1, duration: 0.5 }
			)
		},
	})
}
```
